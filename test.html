const wordPad = document.getElementById('wordPad'); const wordCountDisplay = document.getElementById('wordCount'); const sentenceCountDisplay = document.getElementById('sentenceCount'); const characterLengthDisplay = document.getElementById('characterLength'); function updateCounts() { const text = wordPad.value; const words = text.trim().split(/\s+/).filter(Boolean); const sentences = text.split(/[.!?]\s*/).filter(Boolean); const wordCount = words.length; const sentenceCount = sentences.length; const characterLength = text.length; wordCountDisplay.textContent = `Total word count: ${wordCount}`; sentenceCountDisplay.textContent = `Total sentence count: ${sentenceCount}`; characterLengthDisplay.textContent = `Total character length: ${characterLength}`; } wordPad.addEventListener('input', updateCounts);
